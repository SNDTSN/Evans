//******************************************************************************
//AI토크 관련
//******************************************************************************

//------------------------------------------------------------------------------
//TALK_INTERVAL AI토크 간격 초기치[초]
//------------------------------------------------------------------------------
#globaldefine	TALK_INTERVAL	180


//******************************************************************************
//설치되어 있는 고스트 리스트 구축 관련
//******************************************************************************

//------------------------------------------------------------------------------
//IGLIST_ACCEL
//	1이면 SSP/CROW일 때엔 NOTIFY installedghostname으로
//	설치된 고스트 리스트를 구축하게 됩니다.
//	sakura/kero명 리스트는 작성되지 않지만 빠릅니다.
//	0이면 파일을 체크하여 구축합니다.
//	설치된 고스트가 많으면 시간이 걸립니다.
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.IGLIST_ACCEL	1

//------------------------------------------------------------------------------
//IGLIST_MAX
//	파일 체크로 고스트 리스트를 구축할 때 취득할
//	수의 상한치를 지정합니다. -1이면 제한 없음(전부 취득)입니다.
//	0이면 취득하지 않습니다.
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.IGLIST_MAX	0


//******************************************************************************
//리퀘스트 해석/레스폰스 구축 관련
//******************************************************************************

//------------------------------------------------------------------------------
//SHIORI3FW.REQUEST_LINES_LIMIT
//	리퀘스트 문자열을 해석할 최대 행수입니다.
//	줄이면 기동 등이 빨라지지만 설치된 고스트 리스트를
//	전부 다 가져오지는 못할 수 있습니다.
//	0이면 모든 행을 가져옵니다.
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.REQUEST_LINES_LIMIT	0


//------------------------------------------------------------------------------
//SHIORI3FW.RES_REF_MAX
//	res_reference의 상한
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.RES_REF_MAX			32

//------------------------------------------------------------------------------
//SHIORI3FW.AUTO_DATA_CONVERT
//	1이면 AYA5와 같은 처리를 하게 됩니다.
//	-SAORI의 반환값 등의 자동 타이프 변환/자동적으로 \0x01을 콤마로 하는 등의
//	처리가 들어갑니다.
//	0이면 그러한 처리를 하지 않습니다.
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.AUTO_DATA_CONVERT		1


//------------------------------------------------------------------------------
//SHIORI3FW.REF_ACCEL
//	1이면 reference0 과 같은 변수를 작성하지 않고 reference[0] 와 같은 형태로
//	접근하게 됩니다. 속도가 약간 상승할지도 모릅니다.
//	0이면 reference0 과 같은 변수를 작성합니다 (과거 상호 처리)
//------------------------------------------------------------------------------
#globaldefine	SHIORI3FW.REF_ACCEL				0
